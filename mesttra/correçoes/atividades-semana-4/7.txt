const perCent15 = 0.15
const perCent20 = 0.2

let peso;
let pesoEngordado15;
let pesoEngordado20;


peso = parseFloat(prompt("Informe o peso em KG: "));

pesoEngordado15 = (peso * 1.15);
pesoEngordado20 = (peso * 1.20);

console.log("Caso a pessoa engorde "+ (perCent15 * 100) + "%, ficará com: " + pesoEngordado15.toFixed(2) + " KG");
console.log("Caso a pessoa engorde " + (perCent20 * 100) + "%, ficará com: " + pesoEngordado20.toFixed(2) + " KG");
