let qtde, pQueijo, pPresunto, pHamburger;

qtde = parseInt(prompt("Digite a quantidade de sanduíches: "));

pQueijo = qtde * 50 * 2;
pPresunto = qtde * 50;
pHamburguer = qtde * 120;

console.log(`Para produzir ${qtde} sanduíches serão necessários:`);
console.log(`${(pQueijo / 1000).toFixed(2)} kgs de mussarela`);
console.log(`${(pPresunto / 1000).toFixed(2)} kgs de presunto`);
console.log(`${(pHamburguer / 1000).toFixed(2)} kgs de hambúrguer`);
