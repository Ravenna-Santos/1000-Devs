let qtdePao = parseInt(prompt("Digite a quantidade de pães vendidos: "));
let qtdeBroa = parseInt(prompt("Digite a quantidade de broas vendidas: "));
let valorReforma = parseFloat(prompt("Digite o valor da reforma: "));

let fatPao = qtdePao * 0.12;
let fatBroa = qtdeBroa * 1.60;

let fatDiario = fatPao + fatBroa;
let depositoPop = fatDiario * 0.10;

console.log(`Faturamento com a venda de broas: R$ ${fatBroa.toFixed(2)}`);
console.log(`Faturamento com a venda de pães: R$ ${fatPao.toFixed(2)}`);
console.log(`Faturamento diário (pães + broas): R$ ${fatDiario.toFixed(2)}`);
console.log(`Valor do depósito na poupança: R$ ${depositoPop.toFixed(2)}`);
console.log(`Para pagar a reforma serão necessários: ${Math.ceil(valorReforma / depositoPop)} dias`);
